class Solution {
    int mod = 1000000007;
    public int countGoodNumbers(long n) {
        //ï¼ˆä¸‹æ ‡ä» 0 å¼€å§‹ï¼‰å¶æ•° ä¸‹æ ‡å¤„çš„æ•°å­—ä¸º å¶æ•° ä¸” å¥‡æ•° ä¸‹æ ‡å¤„çš„æ•°å­—ä¸º è´¨æ•° ï¼ˆ2ï¼Œ3ï¼Œ5 æˆ– 7ï¼‰ã€‚
        //å¦‚æœnæ˜¯å¶æ•°ï¼Œé‚£ä¹ˆå„å ä¸€åŠï¼›å¦‚æœnæ˜¯å¥‡æ•°ï¼Œé‚£å¶ä½å¤šä¸€ä¸ª
        long res = myPow(20, n/2);
        if(n % 2 == 1)  res = (res * 5) % mod;
        return (int)res;
    }
    public int myPow(int x, long n) {
        if(x == 0)   return 0;
        long b = n;
        long res = 1;
        if(b < 0){
            x = 1 / x;
            b = -b;
        }
        long y = x;
        while(b > 0){
            if((b & 1) == 1){
                res = res * y % mod;
            }
            b >>= 1;
            y = y*y % mod;
        }
        return (int)res;
    }
}

class Solution {
    public int findLength(int[] A, int[] B) {
        int n = A.length, m = B.length;
        int[][] dp = new int[n + 1][m + 1];
        int ans = 0;
        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                dp[i][j] = A[i] == B[j] ? dp[i + 1][j + 1] + 1 : 0;
                ans = Math.max(ans, dp[i][j]);
            }
        }
        return ans;
    }
}

class Solution {

	public int longestCommonSubpath(int n, int[][] paths) {
		int left = 0, right = 100000;
		while (left < right) {
			int mid = (left + right + 1) / 2;
			if (longestCommonSubpath(paths, mid)) {
				left = mid;
			} else {
				right = mid - 1;
			}
		}
		return left;
	}

	private boolean longestCommonSubpath(int[][] paths, int k) {
		long s = 1000007, ss = 1;
		for (int i = 0; i < k; i++) {
			ss *= s;
		}
		HashMap<Long, Integer> map = new HashMap<>();
		for (int[] path : paths) {
			long hash = 0;
			HashSet<Long> set = new HashSet<>();
			for (int i = 0; i < path.length; i++) {
				hash = (hash * s) + path[i];
				if (i >= k) {
					hash -= path[i - k] * ss;
				}
				if (i + 1 >= k) {
					if (set.add(hash)) {
						map.put(hash, map.getOrDefault(hash, 0) + 1);
						if (map.get(hash) == paths.length) {
							return true;
						}
					}
				}
			}
		}
		return false;
	}
}
